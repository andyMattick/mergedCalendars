Calendar (sample month)

In production, you’ll replace the sample data with a JSON of dates + HS letter extracted from your PDF (e.g., { "date": "2026-08-13", "letter": "A" }).

// --- LocalStorage keys --- const LS_SCHEDULE_KEY = "hsSchedule"; const LS_NOTES_KEY_PREFIX = "hsNotes"; // hsNotes-YYYY-MM-DD // --- State --- let teacherSchedule = {}; // { "1": "AP Stats", ... } let currentModalDay = null; // { date, letter } // --- Helpers --- function loadScheduleFromStorage() { const raw = localStorage.getItem(LS_SCHEDULE_KEY); if (!raw) return; try { teacherSchedule = JSON.parse(raw); } catch (e) { console.warn("Failed to parse stored schedule", e); } } function saveScheduleToStorage() { localStorage.setItem(LS_SCHEDULE_KEY, JSON.stringify(teacherSchedule)); } function getNotesKey(date) { return `${LS_NOTES_KEY_PREFIX}-${date}`; } function loadNotesForDate(date) { const raw = localStorage.getItem(getNotesKey(date)); if (!raw) return {}; try { return JSON.parse(raw); // { periodNumber: "note text" } } catch (e) { console.warn("Failed to parse notes for", date, e); return {}; } } function saveNotesForDate(date, notesObj) { localStorage.setItem(getNotesKey(date), JSON.stringify(notesObj)); } function hasNotesForDate(date) { return !!localStorage.getItem(getNotesKey(date)); } function formatDateLabel(dateStr) { const d = new Date(dateStr); if (isNaN(d)) return dateStr; return d.toLocaleDateString(undefined, { month: "short", day: "numeric", year: "numeric" }); } // --- UI: schedule form --- function hydrateScheduleForm() { for (let p = 1; p <= 7; p++) { const input = document.getElementById(`p${p}`); if (!input) continue; input.value = teacherSchedule[p] || ""; } } function handleScheduleSubmit(event) { event.preventDefault(); const form = event.target; const newSchedule = {}; for (let p = 1; p <= 7; p++) { const input = form[`p${p}`]; newSchedule[p] = input.value.trim() || `Period ${p}`; } teacherSchedule = newSchedule; saveScheduleToStorage(); renderCalendar(); } // --- UI: calendar --- function renderCalendar() { const grid = document.getElementById("calendarGrid"); grid.innerHTML = ""; HS_ROTATION_2026.forEach(day => { const cell = document.createElement("div"); cell.className = "day-cell"; if (hasNotesForDate(day.date)) { cell.classList.add("has-notes"); } const header = document.createElement("div"); header.className = "day-cell-header"; const dateSpan = document.createElement("span"); dateSpan.textContent = formatDateLabel(day.date); const letterSpan = document.createElement("span"); letterSpan.textContent = day.letter; header.appendChild(dateSpan); header.appendChild(letterSpan); cell.appendChild(header); const classesUl = document.createElement("ul"); classesUl.className = "classes-list"; const periods = ROTATION_RULES[day.letter] || []; periods.forEach(periodNum => { const li = document.createElement("li"); const courseName = teacherSchedule[periodNum] || `Period ${periodNum}`; li.textContent = `${periodNum}: ${courseName}`; classesUl.appendChild(li); }); cell.appendChild(classesUl); cell.addEventListener("click", () => openDayModal(day)); grid.appendChild(cell); }); } // --- UI: modal for notes --- function openDayModal(day) { currentModalDay = day; const backdrop = document.getElementById("modalBackdrop"); const dateLabel = document.getElementById("modalDateLabel"); const letterLabel = document.getElementById("modalLetterLabel"); const body = document.getElementById("modalBody"); dateLabel.textContent = formatDateLabel(day.date); letterLabel.textContent = `HS letter: ${day.letter}`; body.innerHTML = ""; const periods = ROTATION_RULES[day.letter] || []; const existingNotes = loadNotesForDate(day.date); periods.forEach(periodNum => { const courseName = teacherSchedule[periodNum] || `Period ${periodNum}`; const block = document.createElement("div"); block.className = "note-block"; const label = document.createElement("label"); label.textContent = `Period ${periodNum} – ${courseName}`; block.appendChild(label); const textarea = document.createElement("textarea"); textarea.value = existingNotes[periodNum] || ""; textarea.dataset.period = periodNum; block.appendChild(textarea); body.appendChild(block); }); backdrop.style.display = "flex"; } function closeDayModal() { const backdrop = document.getElementById("modalBackdrop"); backdrop.style.display = "none"; currentModalDay = null; } function handleSaveNotes() { if (!currentModalDay) return; const body = document.getElementById("modalBody"); const textareas = body.querySelectorAll("textarea"); const notesObj = {}; textareas.forEach(ta => { const period = ta.dataset.period; notesObj[period] = ta.value.trim(); }); saveNotesForDate(currentModalDay.date, notesObj); closeDayModal(); renderCalendar(); // refresh note indicators } function handleClearNotes() { if (!currentModalDay) return; localStorage.removeItem(getNotesKey(currentModalDay.date)); closeDayModal(); renderCalendar(); } // --- Init --- function init() { loadScheduleFromStorage(); hydrateScheduleForm(); renderCalendar(); document.getElementById("scheduleForm") .addEventListener("submit", handleScheduleSubmit); document.getElementById("closeModalBtn") .addEventListener("click", closeDayModal); document.getElementById("saveNotesBtn") .addEventListener("click", handleSaveNotes); document.getElementById("clearNotesBtn") .addEventListener("click", handleClearNotes); } document.addEventListener("DOMContentLoaded", init);
